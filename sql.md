-- CREATE WLD_INV_AUDIT_LOGS TABLE
CREATE TABLE WLD_INV_AUDIT_LOGS (
ID UUID NOT NULL,
ENTITY_NAME VARCHAR(255),
ENTITY_ID VARCHAR(255),
ACTION VARCHAR(255),
OLD_VALUE TEXT,
NEW_VALUE TEXT,
USER_NAME VARCHAR(255),
IP_ADDRESS VARCHAR(255),
REQUEST_ID VARCHAR(255),
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
CONSTRAINT PK_WLD_INV_AUDIT_LOGS PRIMARY KEY (ID)
);

-- CREATE WLD_INV_PERMISSIONS TABLE
CREATE TABLE WLD_INV_PERMISSIONS (
ID UUID NOT NULL,
NAME VARCHAR(255) NOT NULL,
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
UPDATED_AT TIMESTAMP WITHOUT TIME ZONE,
CREATED_BY VARCHAR(255),
UPDATED_BY VARCHAR(255),
CONSTRAINT PK_WLD_INV_PERMISSIONS PRIMARY KEY (ID),
CONSTRAINT UK_WLD_INV_PERMISSIONS_NAME UNIQUE (NAME)
);

-- CREATE WLD_INV_ROLES TABLE
CREATE TABLE WLD_INV_ROLES (
ID UUID NOT NULL,
NAME VARCHAR(255) NOT NULL,
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
UPDATED_AT TIMESTAMP WITHOUT TIME ZONE,
CREATED_BY VARCHAR(255),
UPDATED_BY VARCHAR(255),
CONSTRAINT PK_WLD_INV_ROLES PRIMARY KEY (ID),
CONSTRAINT UK_WLD_INV_ROLES_NAME UNIQUE (NAME)
);

-- CREATE WLD_INV_ROLE_PERMISSIONS JOIN TABLE
CREATE TABLE WLD_INV_ROLE_PERMISSIONS (
ROLE_ID UUID NOT NULL,
PERMISSION_ID UUID NOT NULL,
CONSTRAINT PK_WLD_INV_ROLE_PERMISSIONS PRIMARY KEY (ROLE_ID, PERMISSION_ID),
CONSTRAINT FK_WLD_INV_ROLE_PERMISSIONS_ROLE FOREIGN KEY (ROLE_ID) REFERENCES WLD_INV_ROLES(ID),
CONSTRAINT FK_WLD_INV_ROLE_PERMISSIONS_PERMISSION FOREIGN KEY (PERMISSION_ID) REFERENCES WLD_INV_PERMISSIONS(ID)
);

-- CREATE WLD_INV_USERS TABLE
CREATE TABLE WLD_INV_USERS (
ID UUID NOT NULL,
EMAIL VARCHAR(255) NOT NULL,
FULL_NAME VARCHAR(255),
PASSWORD VARCHAR(255) NOT NULL,
ACTIVE INTEGER,
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
UPDATED_AT TIMESTAMP WITHOUT TIME ZONE,
CREATED_BY VARCHAR(255),
UPDATED_BY VARCHAR(255),
CONSTRAINT PK_WLD_INV_USERS PRIMARY KEY (ID),
CONSTRAINT UK_WLD_INV_USERS_EMAIL UNIQUE (EMAIL)
);

-- CREATE WLD_INV_USER_ROLES JOIN TABLE
CREATE TABLE WLD_INV_USER_ROLES (
USER_ID UUID NOT NULL,
ROLE_ID UUID NOT NULL,
CONSTRAINT PK_WLD_INV_USER_ROLES PRIMARY KEY (USER_ID, ROLE_ID),
CONSTRAINT FK_WLD_INV_USER_ROLES_USER FOREIGN KEY (USER_ID) REFERENCES WLD_INV_USERS(ID),
CONSTRAINT FK_WLD_INV_USER_ROLES_ROLE FOREIGN KEY (ROLE_ID) REFERENCES WLD_INV_ROLES(ID)
);
