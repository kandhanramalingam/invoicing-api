-- CREATE AUDIT_LOGS TABLE
CREATE TABLE AUDIT_LOGS (
ID UUID NOT NULL,
ENTITY_NAME VARCHAR(255),
ENTITY_ID VARCHAR(255),
ACTION VARCHAR(255),
OLD_VALUE TEXT,
NEW_VALUE TEXT,
USER_NAME VARCHAR(255),
IP_ADDRESS VARCHAR(255),
REQUEST_ID VARCHAR(255),
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
CONSTRAINT PK_AUDIT_LOGS PRIMARY KEY (ID)
);

-- CREATE PERMISSIONS TABLE
CREATE TABLE PERMISSIONS (
ID UUID NOT NULL,
NAME VARCHAR(255) NOT NULL,
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
UPDATED_AT TIMESTAMP WITHOUT TIME ZONE,
CREATED_BY VARCHAR(255),
UPDATED_BY VARCHAR(255),
CONSTRAINT PK_PERMISSIONS PRIMARY KEY (ID),
CONSTRAINT UK_PERMISSIONS_NAME UNIQUE (NAME)
);

-- CREATE ROLES TABLE
CREATE TABLE ROLES (
ID UUID NOT NULL,
NAME VARCHAR(255) NOT NULL,
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
UPDATED_AT TIMESTAMP WITHOUT TIME ZONE,
CREATED_BY VARCHAR(255),
UPDATED_BY VARCHAR(255),
CONSTRAINT PK_ROLES PRIMARY KEY (ID),
CONSTRAINT UK_ROLES_NAME UNIQUE (NAME)
);

-- CREATE ROLE_PERMISSIONS JOIN TABLE
CREATE TABLE ROLE_PERMISSIONS (
ROLE_ID UUID NOT NULL,
PERMISSION_ID UUID NOT NULL,
CONSTRAINT PK_ROLE_PERMISSIONS PRIMARY KEY (ROLE_ID, PERMISSION_ID),
CONSTRAINT FK_ROLE_PERMISSIONS_ROLE FOREIGN KEY (ROLE_ID) REFERENCES ROLES(ID),
CONSTRAINT FK_ROLE_PERMISSIONS_PERMISSION FOREIGN KEY (PERMISSION_ID) REFERENCES PERMISSIONS(ID)
);

-- CREATE USERS TABLE
CREATE TABLE USERS (
ID UUID NOT NULL,
EMAIL VARCHAR(255) NOT NULL,
FULL_NAME VARCHAR(255),
PASSWORD VARCHAR(255) NOT NULL,
ACTIVE INTEGER,
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
UPDATED_AT TIMESTAMP WITHOUT TIME ZONE,
CREATED_BY VARCHAR(255),
UPDATED_BY VARCHAR(255),
CONSTRAINT PK_USERS PRIMARY KEY (ID),
CONSTRAINT UK_USERS_EMAIL UNIQUE (EMAIL)
);

-- CREATE USER_ROLES JOIN TABLE
CREATE TABLE USER_ROLES (
USER_ID UUID NOT NULL,
ROLE_ID UUID NOT NULL,
CONSTRAINT PK_USER_ROLES PRIMARY KEY (USER_ID, ROLE_ID),
CONSTRAINT FK_USER_ROLES_USER FOREIGN KEY (USER_ID) REFERENCES USERS(ID),
CONSTRAINT FK_USER_ROLES_ROLE FOREIGN KEY (ROLE_ID) REFERENCES ROLES(ID)
);

-- CREATE EVENTS TABLE
CREATE TABLE EVENTS (
ID UUID NOT NULL,
NAME VARCHAR(255),
DESCRIPTION VARCHAR(255),
DATE TIMESTAMP WITHOUT TIME ZONE,
LOCATION VARCHAR(255),
TYPE VARCHAR(255),
STATUS VARCHAR(255),
DEPOSIT_AMOUNT NUMERIC(19, 2),
SELLER NUMERIC(19, 2),
SELLER_COMMISSION_TYPE BOOLEAN,
SELLER_AUCTION_FEE NUMERIC(19, 2),
SELLER_EVENT_FEE NUMERIC(19, 2),
SELLER_CLUB_FEE NUMERIC(19, 2),
SELLER_VETERINARY_FEE NUMERIC(19, 2),
SELLER_MARKETING_FEE NUMERIC(19, 2),
BUYER NUMERIC(19, 2),
BUYER_COMMISSION_TYPE BOOLEAN,
BUYER_AUCTION_FEE NUMERIC(19, 2),
BUYER_EVENT_FEE NUMERIC(19, 2),
BUYER_CLUB_FEE NUMERIC(19, 2),
BUYER_VETERINARY_FEE NUMERIC(19, 2),
BUYER_MARKETING_FEE NUMERIC(19, 2),
CREATED_AT TIMESTAMP WITHOUT TIME ZONE,
UPDATED_AT TIMESTAMP WITHOUT TIME ZONE,
CREATED_BY VARCHAR(255),
UPDATED_BY VARCHAR(255),
CONSTRAINT PK_EVENTS PRIMARY KEY (ID)
);
